* Behavior
** Mail Location
 #+BEGIN_SRC emacs-lisp
 (setq mu4e-maildir (expand-file-name "~/mail"))
 #+END_SRC
** Shortcuts
 #+BEGIN_SRC emacs-lisp
 (setq mu4e-maildir-shortcuts
       '(("/INBOX"             . ?i)
         ("/[Gmail].Sent Mail" . ?s)
         ("/[Gmail].Trash"     . ?t)))
 #+END_SRC
** Folders
 #+BEGIN_SRC emacs-lisp
 (setq mu4e-drafts-folder "/[Gmail].Drafts")

 (setq mu4e-trash-folder  "/[Gmail].Trash")
 #+END_SRC
